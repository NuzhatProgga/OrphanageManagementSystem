
create database Orphanage
use Orphanage
go



/* Tables */

create table CHILD
(
Child_id varchar(10) primary key,
Child_name varchar(30),
Gender varchar(8),
Date_of_birth varchar(15)
)
go

create table ADOPT
(
Child_id varchar(10),
Parent_name varchar(30),
City varchar(15),
Address	varchar(100),
Phone_number varchar(16),
Adoption_date varchar(15)
constraint child_id_pk primary key(Child_id),
constraint child_id_fk foreign key(Child_id) references CHILD
)




go
create function dbo.CheckFunction(@Child_id varchar(20))/* Function */
returns int
as begin
	IF EXISTS(SELECT 1 FROM ADOPT WITH(NOLOCK)
          WHERE Child_id = @Child_id)
			return (select 1)
	ELSE
		return(select 0)

return -1
end/* Function */
go






create table DEPARTMENT
(
Dept_name nvarchar(20),
Dept_id varchar(10),
constraint dept_name_id_PK primary key (Dept_name,Dept_id)
)
go

create table EMPLOYEE
(
Employee_id varchar(10) primary key,
Employee_name nvarchar(30),
Gender varchar(8),
Dept_name nvarchar(20),
Dept_id varchar(10),
Salary int,
constraint dept_name_id_FK foreign key (Dept_name,Dept_id) references DEPARTMENT
)
go

create table ADVISOR
(
Child_id varchar(10) primary key,
Advisor_id varchar(10) not null,
foreign key(Child_id) references CHILD,
foreign key(Advisor_id) references EMPLOYEE,
check (dbo.CheckFunction(Child_id )=0),
check(Advisor_id like 'Gd%')
)
go

create table DONOR(
Donor_id varchar(10) primary key,
Donor_name nvarchar(30) ,
City nvarchar(15),
Address varchar(100),
Phone_number varchar(16) unique
)
go

create table DONATION(
Donor_id varchar(10) ,
Employee_id varchar(10),
Date varchar (16),
Amount int,
check(Employee_id like 'Acc%'),
primary key(Donor_id,Employee_id,Date),
foreign key(Donor_id) references DONOR,
foreign key(Employee_id) references EMPLOYEE
)
go


create table BUILDING(
Building_name varchar(20),
Floor tinyint,
Room_no tinyint,
Primary key(Building_name,Floor,Room_no)
)
go



create table RESIDENT(
Child_id varchar(10) unique,
Building_name varchar(20),
Floor tinyint,
Room_no tinyint,
Bed_no tinyint,
check (dbo.CheckFunction(Child_id )=0) ,
Primary key(Building_name,Floor,Room_no,bed_no),
foreign key(Child_id) references CHILD
)
go


create table TEACHER(
Teacher_id varchar(10),
Teacher_name nvarchar(30),
Salary int,
constraint teacher_id_PK primary key(Teacher_id)
)
go

create table SCHOOL(
Child_id varchar(10) primary key,
Class tinyint,
Class_teacher_id varchar(10),
check (dbo.CheckFunction(Child_id )=0) ,
check(class<=10),
foreign key(Child_id) references CHILD,
foreign key(Class_teacher_id) references TEACHER
)
go
